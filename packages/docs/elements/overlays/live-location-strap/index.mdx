## Usage

Place the strap inside a `<Sequence>` and set only the fields you need:

```tsx title="MyComposition.tsx"
import {Sequence} from 'remotion';
import {LiveLocationStrap} from './live-location-strap';

export const MyComposition = () => {
  return (
    <Sequence from={30} durationInFrames={120} layout="none">
      <LiveLocationStrap
        statusLabel="On location"
        location="NAB Show"
        venue="Main Stage"
        date="June 22"
      />
    </Sequence>
  );
};
```
