from moviepy.editor import *
import numpy as np

# Dirk pink background    
bg = ColorClip(size=(720,1280), color=(30,0,50)).set_duration(10)

# Sneha Glow Text
name = TextClip("✨ SNEHA ✨", fontsize=170, color="pink", font="Arial-Bold")
name = name.set_position("center").set_duration(10).fadein(2).fadeout(2)

# Love Message
msg = TextClip("Forever In My Heart ❤️", fontsize=70, color="white", font="Arial-Bold")
msg = msg.set_position(("center", 1000)).set_duration(10)

# Floating Heart Effect
hearts = []
for i in range(30):
    heart = TextClip("💗", fontsize=50)
    heart = heart.set_position((np.random.randint(0,720),
                                 np.random.randint(200,1200)))
    heart = heart.set_duration(0.3).set_start(i*0.3)
    hearts.append(heart)

video = CompositeVideoClip([bg, name, msg, *hearts])
video.write_videofile("sneha_pro_intro.mp4", fps
