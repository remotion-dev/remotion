## Example

```tsx twoslash title="src/Spark.tsx"
import { Spark } from "@remotion/shapes";
import { AbsoluteFill } from "remotion";

  return (

      style={{

    >
      <Spark
        width={240}
        height={336}
        edgeRoundness={1}

      />
    </AbsoluteFill>
  );
};
```
